import * as React from "react";
import Table from "@mui/material/Table";
import TableBody from "@mui/material/TableBody";
import TableCell from "@mui/material/TableCell";
import TableContainer from "@mui/material/TableContainer";
import TableHead from "@mui/material/TableHead";
import TableRow from "@mui/material/TableRow";
import Paper from "@mui/material/Paper";
import { useStyles, BasicTableStyle } from "./BasicTable.styles";
 
import CatlogImg from "./CatalogImg/CatlogImg";
import Buttons from "./Buttons/Buttons";
 
export default function BasicTable(props) {
  const classes = useStyles();
  let data = props?.data;
  let rows = [];
 
  const getInnerFirstRow = ({
    values,
    data,
    imageRowsMerge = null,
    key,
    rowHeadRowsMerge = null,
  }) => {
    let rowValues = [];
    let columns = values?.amount.length;
 
    for (let j = 0; j < columns; j++) {
      let data = values?.amount[j]?.amountValue[0];
      rowValues.push(<TableCell key={`${0} ${j}`}>{data}</TableCell>);
    }
 
    return (
      <TableRow
        key={key}
        sx={{ "&:last-child td, &:last-child th": { border: 0 } }}
      >
        {imageRowsMerge && (
          <TableCell component="th" scope="row" rowSpan={imageRowsMerge}>
            <div className={classes.priceCard}>
              <CatlogImg data={data}></CatlogImg>
            </div>
          </TableCell>
        )}
 
        <TableCell rowSpan={rowHeadRowsMerge}>{values.campaignName}</TableCell>
 
        {rowValues}
 
        <TableCell rowSpan={rowHeadRowsMerge}>
          <Buttons data={{ product: values, commonData: data }}></Buttons>
        </TableCell>
      </TableRow>
    );
  };
 
  data.forEach((product, prodIndex) => {
    let productCampaign = product.campaign;
    let imgRowToMerge = 0;
    productCampaign.forEach((subRows) => {
      imgRowToMerge +=
        subRows?.amount?.length && subRows?.amount[0]?.amountValue?.length;
    });
 
    productCampaign.forEach((subRows, subIndex) => {
      let amount = subRows?.amount;
      let rowHeadRowsMerge = subRows?.amount[0]?.amountValue?.length;
 
      rows.push(
        getInnerFirstRow({
          key: `${prodIndex} ${subIndex} 1`,
          values: subRows,
          data: product,
          imageRowsMerge: subIndex === 0 ? imgRowToMerge : null,
          rowHeadRowsMerge,
        })
      );
      for (let i = 1; i < rowHeadRowsMerge; i++) {
        let rowValues = [];
        let columns = subRows?.amount.length;
        for (let j = 0; j < columns; j++) {
          let data = subRows?.amount[j]?.amountValue[i];
          rowValues.push(<TableCell key={`${i} ${j}`}>{data}</TableCell>);
        }
        rows.push(
          <TableRow
            key={`${prodIndex} ${subIndex} ${i + 1}`}
            sx={{ "&:last-child td, &:last-child th": { border: 0 } }}
          >
            {rowValues}
          </TableRow>
        );
      }
    });
  });
  return (
    <TableContainer component={Paper}>
      <Table sx={{ minWidth: 650 }} aria-label="simple table">
        <TableHead>
          <TableRow>
            <TableCell>Empty</TableCell>
 
            <TableCell>Compaign</TableCell>
 
            <TableCell>Special-Priced Device</TableCell>
            <TableCell>Advance service payment</TableCell>
            <TableCell>Starter Package</TableCell>
 
            <TableCell>Channels</TableCell>
          </TableRow>
        </TableHead>
        <TableBody>{rows}</TableBody>
      </Table>
    </TableContainer>
  );
}
