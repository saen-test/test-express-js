import React from "react";
import { Userstyled, CarouselCardSpacing } from "./RecommendedCarousal.style";
import { firstCarousel } from "../../AuthoredData/HotDeal/RecommendedCarousalData";
import { listOfAllProductSelector, isLoadingSelector } from "../../API/HotDealComponent/HotDealSlice";
import hotDeals from "../../assets/images/phone.png";
import Carousel from '../common/Carousel/Carousel';
import { useSelector } from "react-redux";
import { Box, CardMedia, Avatar, Grid } from "@mui/material";
import NetworkFlag from "../../assets/images/NetworkCompatibilityFlag.png";
import { useNavigate } from "react-router-dom";
 
const DesktopCarousal = (props) => {
    const { isSecondCarousal = false } = props || {};
    const list = useSelector(listOfAllProductSelector);
    const isLoading = useSelector(isLoadingSelector);
    const classes = Userstyled();
    const navigate = useNavigate();
    const {
        date = "",
        headingText = "",
        description2 = "",
        description1 = "",
        viewAllText = "",
        startsFromLabel = "",
        devicePriceLabel = "",
        selectButtonText = "",
        selectButtonLink = "",
        backgroundImage = '',
    } = props?.props || firstCarousel ;
 
    const carouselConfig = {
        autoPlay: false,
        swipeable: true,
        draggable: false,
        showDots: true,
        arrows: false,
        keyBoardControl: true,
        customTransition: "all .5",
        containerClass: "carousel-container",
        removeArrowOnDeviceType: ["tablet", "mobile"],
        dotListClass: "custom-dot-list-style",
        itemClass: "carousel-item-padding-40-px",
        slidesToSlide: 1,
        height: 400,
        transitionDuration: 300,
        shouldResetAutoplay: false,
        responsive: {
            superLargeDesktop: {
                breakpoint: { max: 4000, min: 3000 },
                items: 5
            },
            desktop: {
                breakpoint: { max: 3000, min: 1024 },
                items: 3
            },
            tablet: {
                breakpoint: { max: 1024, min: 464 },
                items: 2
            },
            mobile: {
                breakpoint: { max: 464, min: 0 },
                items: 1
            }
        }
    };
 
 
 
    let cartData = '';
    if (list.length > 0) {
        cartData =
            list?.map((item) => {
                return (
                    <div className={classes.cardOuter}>
                        <Box className="card-media-container" onClick={() => alert("PDP")}>
                            <Box className={classes.cardHeader}>
                                <span className={classes.cardRecommend} label="Recommended">Recommended</span>
                                <Avatar className="network-logo" src={NetworkFlag} />
                            </Box>
                            <CardMedia className={classes.cardImg} component="img" image={hotDeals} alt="product image" />
                        </Box>
 
                        <div className={classes.cardContent}>
                            <div className={classes.phoneModel}>
                                {item.name}
                            </div>
                            <div className={classes.phoneStorage}>256GB / 512GB</div>
                        </div>
                        <div className={classes.cardContentFooter}>
                            <div className={classes.cardConetntBelow}>
                                <div className={classes.price}> {item?.price_range?.minimum_price?.regular_price?.currency} {item?.price_range?.minimum_price?.regular_price?.value}</div>
                                <div className={classes.ogPrice}>{item?.price_range?.minimum_price?.regular_price?.currency} {item?.price_range?.minimum_price?.regular_price?.value}</div>
                            </div>
                            <button onClick={() => {
                                navigate(selectButtonLink);
                                window.location.reload();
                            }} className={classes.cardButton}>{selectButtonText}</button>
                        </div>
                    </div>
                )
 
            });
    }
    
    return (
        <>
            {!isLoading && cartData !== '' ?
                <div className={isSecondCarousal ? classes.mainDivSecondCarousal : classes.mainDiv} style={{'background-image':`url(${backgroundImage})`}}>
                    <Grid container spacing={2} columns={16}
                    >
                        <Grid item xs={4}>
                            <div className={classes.leftDiv}>
                                <div className={classes.leftUpperDiv}>
                                    <div className={classes.leftUpperContent}>
                                        {!isSecondCarousal && <div className={classes.leftUpperdate}>{date}</div>}
                                        <div className={isSecondCarousal ? classes.leftUppertitleTextSecondCarousal : classes.leftUppertitleText}>{headingText}</div>
                                        <div className={isSecondCarousal ? classes.leftUpperdescriptionSecondCarousal : classes.leftUpperdescription}>{description1}</div>
                                    </div>
                                </div>
 
                                <div className={classes.leftLowerDiv}>
                                    <div className={classes.leftLowerContent}>
                                        <div className={isSecondCarousal ? classes.leftLowerstartsFromLabelSecondCarousal : classes.leftLowerstartsFromLabel}>{startsFromLabel}</div>
                                        <div className={isSecondCarousal ? classes.leftLowerstartsFromPriceSecondCarousal : classes.leftLowerstartsFromPrice}>{devicePriceLabel}</div>
                                        <div className={isSecondCarousal ? classes.leftLowerdevicePriceLabeleSecondCarousal : classes.leftLowerdevicePriceLabel}>{description2}</div>
                                        <button className={classes.leftLowerviewAllButtonText}>{viewAllText}</button>
                                    </div>
                                </div>
 
                            </div>
                        </Grid>
                        <Grid item xs={12}>
                            <CarouselCardSpacing>
 
                                <Carousel
                                    config={carouselConfig}
                                >
                                    {!isLoading ? cartData : ''}
 
                                </Carousel>
                            </CarouselCardSpacing>
 
                        </Grid>
                    </Grid>
                </div>
                : ''}
        </>
    )
};
export default DesktopCarousal;
